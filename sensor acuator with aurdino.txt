#include <LiquidCrystal.h>

float temp;
int sensor = A0;
float tempc;
float tempf;

// initialize the library with the numbers of the interface pins
LiquidCrystal lcd(2, 3, 4, 5, 6, 7);

void setup()
{
  lcd.begin(16, 2); // set up the LCD's number of columns and rows
}

void loop()
{
  temp = analogRead(sensor); // Read analog voltage from sensor
  tempc = (temp * 4.88) / 10; // Convert ADC value to temperature in °C
  tempf = (tempc * 1.8) + 32; // Convert °C to °F

  lcd.clear(); // clear LCD before displaying new values

  lcd.setCursor(0, 0);
  lcd.print("Temp in C = ");
  lcd.print(tempc); // Use print(), not println() for LCD

  lcd.setCursor(0, 1);
  lcd.print("Temp in F = ");
  lcd.print(tempf); // Use print() here too

  delay(1000); // wait 1 second before next update
}
