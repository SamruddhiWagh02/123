import RPi.GPIO as GPIO     # Library for Raspberry Pi GPIOs
import time                 # Library to use sleep function

# Use BCM pin numbering
GPIO.setmode(GPIO.BCM)

# Define GPIO pins for Trigger and Echo
GPIO_TRIGGER = 12
GPIO_ECHO = 13

# Set up GPIO directions
GPIO.setup(GPIO_TRIGGER, GPIO.OUT)  # Trigger pin as Output
GPIO.setup(GPIO_ECHO, GPIO.IN)      # Echo pin as Input


# Function to calculate distance
def distance():
    # Send a short 10 µs pulse to trigger
    GPIO.output(GPIO_TRIGGER, True)
    time.sleep(0.00001)  # 10 microseconds
    GPIO.output(GPIO_TRIGGER, False)

    # Record the last low timestamp
    StartTime = time.time()
    StopTime = time.time()

    # Save the start time (when Echo goes HIGH)
    while GPIO.input(GPIO_ECHO) == 0:
        StartTime = time.time()

    # Save the arrival time (when Echo goes LOW)
    while GPIO.input(GPIO_ECHO) == 1:
        StopTime = time.time()

    # Calculate time difference
    TimeElapsed = StopTime - StartTime
    # Distance = (Speed of sound × Time) / 2
    dist = (TimeElapsed * 34300) / 2  # in cm

    return dist


# Main program
if __name__ == '__main__':
    try:
        while True:
            dist = distance()  # Call distance function
            print("Measured Distance = %.1f cm" % dist)
            time.sleep(2)  # Wait 2 seconds before next measurement

    # Stop with Ctrl + C
    except KeyboardInterrupt:
        print("Measurement stopped by User")
        GPIO.cleanup()  # Release GPIO resources
