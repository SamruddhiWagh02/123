int led1 = 5;
int led2 = 6;
int led3 = 7;
char data; // should be char, not int (since we receive characters)

void setup()
{
  Serial.begin(9600); // Start Bluetooth serial communication
  pinMode(led1, OUTPUT);
  pinMode(led2, OUTPUT);
  pinMode(led3, OUTPUT);
}

void loop()
{
  while (Serial.available() > 0)
  {
    data = Serial.read();   // Read data from Bluetooth
    Serial.println(data);   // Echo back to Serial Monitor (optional)

    if (data == 'A')
      digitalWrite(led1, HIGH); // Turn ON LED1

    if (data == 'B')
      digitalWrite(led1, LOW);  // Turn OFF LED1

    if (data == 'C')
      digitalWrite(led2, HIGH); // Turn ON LED2

    if (data == 'D')
      digitalWrite(led2, LOW);  // Turn OFF LED2

    if (data == 'E')
      digitalWrite(led3, HIGH); // Turn ON LED3

    if (data == 'F')
      digitalWrite(led3, LOW);  // Turn OFF LED3
  }
}
